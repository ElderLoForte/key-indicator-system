 17 /**
 18  * Returns the areaID string of the given area.
 19  * @param {*} allSheetData 
 20  * @param {string} areaName 
 21  * @returns {string} The areaID string.
 22  */
 23 function getAreaID(allSheetData, areaName) {
 24     let cache = CacheService.getDocumentCache();
 25     let areaIDs_JSONString = cache.get(AREA_IDS_CACHE_KEY);
 26 
 27     let areaIDs = areaIDs_JSONString == null
 28         ? loadAreaIDs(allSheetData)
 29         : JSON.parse(areaIDs_JSONString);
 30 
 31     if (typeof areaIDs[areaName] == 'undefined')
 32         throw new ReferenceError(`Unable to get areaID - couldn't find area '${areaName}'`);
 33 
 34     return areaIDs[areaName];
 35 }

 41 function createTemplates_(filesystemObject, templateID) {
 42     // creates per-zone templates, moves them into the correct folders, and then
 43     // returns a modified filesystemObject for working with later on down the line.
 44     // if the template already exists, it leaves it alone and moves along.
 45     let filesystemObjectCopy = filesystemObject;
 46     // Logger.log([[[[[[[filesystemObjectCopy]]]]]]])
 47     let templateFile = DriveApp.getFileById(templateID);
 48 
 49     for (let i = 0; i < filesystemObjectCopy.name.length; i++) {
 50         if (filesystemObjectCopy.docID[i] == "Doc Id" || filesystemObjectCopy.docID[i] == "DOC ID" || filesystemObjectCopy.docID[i] == "") {
 51 
 52             // if(functionGUBED == true){Logger.log([i,filesystemObjectCopy.parentFolderID[i],filesystemObjectCopy.folderID[i]])}
 53             let targetFolder = filesystemObjectCopy.folderID[i];
 54             // if(functionGUBED == true){Logger.log(filesystemObjectCopy.folderID[i])}
 55             let folderObject = DriveApp.getFolderById(targetFolder);
 56             let fileName = filesystemObjectCopy.name[i];
 57             let templateCopy = templateFile.makeCopy(fileName, folderObject);
 58             filesystemObjectCopy.docID[i] = templateCopy.getId();
 59         }
 60     }
 61     return filesystemObjectCopy;
 62 }

117 function modifyTemplates_(filesystemObject, referenceDataSheet, scope) {

130     let scopeString = "";
131     switch (scope) {
132         case reportLevel.area:

139             break;

142             columnPosition = kicHeader.indexOf("Zone");
143             break;
144     }
145     Logger.log("pre-Split");
146     let isDuplicateColumnPosition = kicHeader.indexOf("isDuplicate");

160 
161     for (let i = 0; i < filesystemObject.name.length; i++) {

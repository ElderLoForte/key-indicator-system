  3 class FilesystemEntry {
  4     constructor(name, parentFolderID, folderID, sheetReportID1, sheetReportID2) {
  5         this.folderName = name;
  6         this.parentFolder = parentFolderID;
  7         this.folder = folderID;
  8         this.sheetID1 = sheetReportID1;
  9         this.sheetID2 = sheetReportID2;
 10         // updated on 12/28/2021 to bring into the same format as the setup in sheetData for filesys entries
 11     }
 12     toArray() {
 13         return [this.folderName, this.parentFolder, this.folder, this.sheetID1, this.sheetID2];
 14     }
 15 
 16     toArrayFancy(sheetData) {
 17         // this one gives data that matches up with the SheetData order
 18         // thus making it 100% compliant with changes that Elder Gerlek may make in the future to the formatting.
 19         let outArray = [];
 20         outArray[sheetData.getIndex("folderName")] = this.folderName;
 21         outArray[sheetData.getIndex("parentFolder")] = this.parentFolder;
 22         outArray[sheetData.getIndex("folder")] = this.folder;
 23         outArray[sheetData.getIndex("sheetID1")] = this.sheetID1;
 24         outArray[sheetData.getIndex("sheetID2")] = this.sheetID2;
 25         return outArray;
 26     }
 27 }
 28 
 29 /*
 30       "folderName": 0,
 31       "parentFolder":1,
 32       "folder":2,
 33       "sheetID1":3,
 34       "sheetID2":4,
 35 */
 36 
 37 function getDataFromArray_(fsObjArray, sheetData) {
 38     let outData = [];
 39     for (let entry of fsObjArray) {
 40         outData.push(entry.toArrayFancy(sheetData));
 41     }
 42     return outData;
 43 }
 44 
 45 function loadFSIntoClass_(data, sheetData) {
 46     // this function and the toArray function will need to be changed if the column order gets changed.
 47     let fsData = [];
 48 
 49     for (let item of data) {
 50         // this is a basic loader doodad, it can become more smart in the future if I want it to by incorporating Elder Gerlek's sheetloader indexOf thingy.
 51         let entry = new FilesystemEntry(item[sheetData.getIndex("folderName")], item[sheetData.getIndex("parentFolder")], item[sheetData.getIndex("folder")], item[sheetData.getIndex("sheetID1")], item[sheetData.getIndex("sheetID2")]);
 52         fsData.push(entry);
 53     }
 54 
 55     return fsData;
 56 }

  4 function roleParser(labelNameObject, listOfEmails) {
  5     let compRoles = [];
  6     let isTrainer = [];
  7     // Position Strings: First is the tested for value, second is the readable output one.
  8     let ap = ["(AP)", "AP"];
  9     let zl1 = ["(ZL1)", "ZL1"];
 10     let zl2 = ["(ZL2)", "ZL2"];
 11     let zl3 = ["(ZL3)", "ZL3"];
 12     let stl1 = ["(STL1)", "STL1"];
 13     let stl2 = ["(STL2)", "STL2"];
 14     let stl3 = ["(STL3)", "STL3"];
 15     let dl = ["(DL)", "DL"];
 16     let seniorComp = ["(SC)", "SC"];
 17     let juniorComp = ["(JC)", "JC"];
 18     let seniorCompTrainer = ["(TR)", "TR"];
 19     let zlTrainer = ["(ZLT)", "ZLT"];
 20     let stlTrainer = ["(STLT)", "STLT"];
 21     let dlTrainer = ["(DT)", "DT"];
 22     let specialAssignment = ["(SA)", "SA"];
 23 
 24 
 25 
 26     // Logger.log([labelNameObject[0],labelNameObject[1],labelNameObject[2],labelNameObject[3]])
 27     for (let labelIterant = 0; labelIterant < listOfEmails.length; labelIterant++) {
 28         let labelString = labelNameObject[labelIterant].toString();
 29         if (typeof labelNameObject[labelIterant] == "object") {
 30             Logger.log(["weird edge-case thingy:", labelIterant, listOfEmails[labelIterant], labelNameObject[labelIterant], labelNameObject[labelIterant].toString()]);
 31         }
 32         let openParenPos = labelString.lastIndexOf("("); /* indexOf was originally used here, switched to lastIndexOf because somebody */
 33         let closeParenPos = labelString.lastIndexOf(")"); /* with a nickname in parenthesis, ie (Siri) was breaking it and they weren't getting their role parsed properly. */
 34         /* https://www.codegrepper.com/code-examples/javascript/find+the+last+occurrence+of+a+character+in+a+string+javascript */
 35         // let regexSearch = labelString.search(/\(\w*/) // this one stopped working and was harder to debug and read, so I switched it.
 36         let roleString = labelString.substring(openParenPos, closeParenPos + 1);
 37         switch (roleString) {
 38             // AP's
 39             case ap[0]:
 40                 compRoles[labelIterant] = ap[1];
 41                 isTrainer[labelIterant] = false;
 42                 break;
 43             // ZL
 44             case zl1[0]:
 45                 compRoles[labelIterant] = zl1[1];
 46                 isTrainer[labelIterant] = false;
 47                 break;
 48             case zl2[0]:
 49                 compRoles[labelIterant] = zl2[1];
 50                 isTrainer[labelIterant] = false;
 51                 break;
 52             case zl3[0]:
 53                 compRoles[labelIterant] = zl3[1];
 54                 isTrainer[labelIterant] = false;
 55                 break;
 56             case zlTrainer[0]:
 57                 compRoles[labelIterant] = zlTrainer[1];
 58                 isTrainer[labelIterant] = true;
 59                 break;
 60             //STL
 61             case stl1[0]:
 62                 compRoles[labelIterant] = stl1[1];
 63                 isTrainer[labelIterant] = false;
 64                 break;
 65             case stl2[0]:
 66                 compRoles[labelIterant] = stl2[1];
 67                 isTrainer[labelIterant] = false;
 68                 break;
 69             case stl3[0]:
 70                 compRoles[labelIterant] = stl3[1];
 71                 isTrainer[labelIterant] = false;
 72                 break;
 73             case stlTrainer[0]:
 74                 compRoles[labelIterant] = stlTrainer[1];
 75                 isTrainer[labelIterant] = true;
 76                 break;
 77 
 78             // District Leader
 79             case dl[0]:
 80                 compRoles[labelIterant] = dl[1];
 81                 isTrainer[labelIterant] = false;
 82                 break;
 83             case dlTrainer[0]:
 84                 compRoles[labelIterant] = dlTrainer[1];
 85                 isTrainer[labelIterant] = true;
 86                 break;
 87             // Senior Comp
 88             case seniorComp[0]:
 89                 compRoles[labelIterant] = seniorComp[1];
 90                 isTrainer[labelIterant] = false;
 91                 break;
 92             case seniorCompTrainer[0]:
 93                 compRoles[labelIterant] = seniorCompTrainer[1];
 94                 isTrainer[labelIterant] = true;
 95                 break;
 96             // Junior Comp
 97             case juniorComp[0]:
 98                 compRoles[labelIterant] = juniorComp[1];
 99                 isTrainer[labelIterant] = false;
100                 break;
101             // Special Assignment
102             case specialAssignment[0]:
103                 compRoles[labelIterant] = specialAssignment[1];
104                 isTrainer[labelIterant] = false;
105                 break;
106         }
107     }
108     return {
109         compRoles: compRoles,
110         isTrainer: isTrainer
111     };
112 } // Written by Elder Robertson, TMM

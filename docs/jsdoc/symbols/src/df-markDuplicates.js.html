  9 /**
 10  * Flags duplicate responses in the Data sheet. Old version to be replaced.
 11  * @param {*} allSheetData 
 12  */
 13 function markDuplicates_old(allSheetData) { //                                  TODO: Don't pull the whole sheet?
 14     Logger.log("Marking duplicate responses. Pulling data...");
 15     Logger.log("TODO: Don't pull the whole sheet?");
 16 
 17     let sd = allSheetData.data;
 18     let sheet = sd.getSheet();
 19     let vals = sheet.getDataRange().getValues();
 20 
 21     let mostRecentResponse = {};   //Previously checked responses. Uses rIDs as keys, and contains row numbers for marking and timestamps for comparison
 22     let duplicates = [];  //Row indices that contain duplicates
 23     let skippedRows = []; //Row indices that are blank
 24 
 25     let minRow; //Used for bounds on the edit range
 26     let maxRow = vals.length - 1;   //Used for bounds on the edit range
 27 
 28     //WARNING: this assumes header is on the top row
 29     if (CONFIG.MARK_DUPLICATES_MAX_ROW_TO_CHECK > -1)
 30         maxRow = Math.min(maxRow, CONFIG.MARK_DUPLICATES_MAX_ROW_TO_CHECK);
 31 
 32 
 33 
 34 
 35     let firstPass = true; //Used to make the loop run twice (in case it's not sorted by timestamp)
 36     console.info("TODO: make mark dupes loop run twice");
 37 
 38     for (let row = maxRow; row > 0; row--) {    //WARNING: this assumes header is on the top row
 39 
 40         /*if (row == maxRow) {
 41             console.time("Time to process 100 lines");
 42         } else if (row % 100 == 0 && row > 100) {
 43             console.timeEnd("Time to process 100 lines");
 44             console.time("Time to process 100 lines");
 45         } else {
 46             console.timeEnd("Time to process 100 lines");
 47         }*/
 48 
 49         let log = "Checking if row index " + (row + 1) + " is a duplicate...";
 50 
 51         //Skip empty rows
 52         if (vals[row][sd.getIndex('areaName')] == "") {
 53             log += '\nSkipping row';
 54             skippedRows.push(row);
 55 
 56             if (DBCONFIG.LOG_DUPLICATES) Logger.log(log);
 57             continue;
 58         }
 59 
 60 
 61         minRow = row; //Update topmost row
 62 
 63         let areaName = vals[row][sd.getIndex('areaName')];
 64         let areaID;
 65         try {
 66             areaID = getAreaID(allSheetData, areaName);
 67         } catch (e) {
 68             console.warn("Couldn't get areaID on line " + (row + 1) + " while marking duplicates. Area '" + areaName + "' not found");
 69         }
 70 
 71 
 72         let kiDate = vals[row][sd.getIndex('kiDate')];
 73         let tstamp = vals[row][sd.getIndex('formTimestamp')];
 74 
 75         let rID = areaID + " | " + kiDate;   //Defined such that duplicate responses should have identical response IDs
 76 
 77         log += "\nResponse ID: '" + rID + "'";
 78 
 79         if (typeof mostRecentResponse[rID] == 'undefined') { //If this is the first ocurrence, add to mostRecentResponse and skip
 80             mostRecentResponse[rID] = { "tstamp": tstamp, "row": row };
 81 
 82             log += "\nFirst ocurrence, continuing.";
 83             if (DBCONFIG.LOG_DUPLICATES) Logger.log(log);
 84             continue;
 85         }
 86 
 87         let prev = mostRecentResponse[rID];  //Previous ocurrence
 88         log += "\nFound previous ocurrence on row index ${prev.row}, comparing...";
 89 
 90         //Handle comparing historical records (which don't have timestamps)
 91         if (typeof prev.tstamp == "string") {
 92 
 93             if (typeof tstamp == "string")
 94                 log += "\nBoth ocurrences are historical records, can't determine which to keep. Keeping the current one and marking previous as a duplicate";
 95             else
 96                 log += "\nPrev ocurrence is a historical record. Keeping the current one and marking previous as a duplicate";
 97 
 98             duplicates.push(prev.row);
 99             mostRecentResponse[rID] = { "tstamp": tstamp, "row": row };
100         }
101         else if (typeof tstamp == "string") {
102             log += "Current ocurrence is a historical record, but the previous one is not. Keeping the previous and marking the current as a duplicate";
103             duplicates.push(row);
104         }
105         else  //Handle non-historical responses
106         {
107             //If this is more recent than the previous ocurrence
108             if (tstamp.getTime() > prev.tstamp.getTime()) {
109                 log += "\nThe current response is more recent, replacing and marking the old as duplicate.";
110 
111                 duplicates.push(prev.row);
112                 mostRecentResponse[rID] = { "tstamp": tstamp, "row": row };
113             }
114             else {
115                 log += "\nThe previous response is more recent, marking the current as duplicate.";
116                 duplicates.push(row);
117             }
118         }
119 
120         if (DBCONFIG.LOG_DUPLICATES) Logger.log(log);
121     }
122 
123     Logger.log("Finished pulling duplicate data. Pushing to sheet...");
124 
125 
126     let out = [];
127 
128     for (let row = minRow; row <= maxRow; row++) {
129         let isDuplicate = duplicates.includes(row);
130         let isSkipped = skippedRows.includes(row);
131         out[row - minRow] = [];
132         out[row - minRow][0] = isDuplicate ? true :
133             isSkipped ? "" : false;
134     }
135 
136     sheet.getRange(minRow + 1, sd.getIndex('isDuplicate') + 1, out.length, 1).setValues(out);
137 
138 
139     Logger.log("Finished marking duplicate responses.");
140 
141 }

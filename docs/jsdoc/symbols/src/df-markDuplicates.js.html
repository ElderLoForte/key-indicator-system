 19     let duplicates = [];  //Row indices that contain duplicates
 20     let skippedRows = []; //Row indices that are blank

 26     if (CONFIG.MARK_DUPLICATES_MAX_ROW_TO_CHECK > -1)

 70         let tstamp = vals[row][sd.getIndex('formTimestamp')];

 82         }
 83

 85         log += `\nFound previous ocurrence on row index ${prev.row}, comparing...`;

 94 
 95             duplicates.push(prev.row);

 97         }
 98         else if (typeof tstamp == "string") {
 99             log += `Current ocurrence is a historical record, but the previous one is not. Keeping the previous and marking the current as a duplicate`;
100             duplicates.push(row);
101         }
102         else  //Handle non-historical responses

104             //If this is more recent than the previous ocurrence

106                 log += `\nThe current response is more recent, replacing and marking the old as duplicate.`;

108                 duplicates.push(prev.row);
109                 mostRecentResponse[rID] = { "tstamp": tstamp, "row": row };
110             }
111             else {
112                 log += `\nThe previous response is more recent, marking the current as duplicate.`;
113                 duplicates.push(row);
114             }
115         }

118     }

129         out[row - minRow][0] = isDuplicate ? true :

  6 /**
  7  * Shares folders in the file structure with the appropriate area emails.
  8  * Pulls contact data from the Contacts sheet and file structure data from the Filesys sheets.
  9  */
 10 function shareFileSys() {
 11     Logger.log(`Beginning read/write file sharing...`);
 12 
 13     let allSheetData = constructSheetData();
 14     let contacts = getContactData(allSheetData);
 15     let missionOrgData = getMissionLeadershipData(contacts);
 16 
 17     //List of emails with full access
 18     let officeEmails = [];
 19     officeEmails.push(missionOrgData.apArea.areaEmail); //Add AP email
 20     if (missionOrgData.hasStltArea) officeEmails.push(missionOrgData.stltArea.areaEmail); //Add STLT email if it exists
 21 
 22 
 23 
 24 
 25     //Pull and sort area, dist, zone file data
 26 
 27 
 28     function getFolders(sheetData) {
 29         let allFoldersList = sheetData.getData(); //List of objects containing the folder data for each folder
 30         let allFolders = {};                      //Turn allFoldersList into an object keyed by name, rather than a list
 31         //i.e. allZoneFolders["BROWNSVILLE"] returns Brownsville zone's folder data
 32         for (let i = 0; i < allFoldersList.length; i++) {
 33             let folder = allFoldersList[i];
 34             let name = folder.folderName;
 35             allFolders[name] = folder;
 36         }
 37         return allFolders;
 38     }
 39 
 40     let allZoneFolderData = getFolders(allSheetData.zoneFilesys);
 41     let allDistFolderData = getFolders(allSheetData.distFilesys);
 42     let allAreaFolderData = getFolders(allSheetData.areaFilesys);
 43 
 44 
 45 
 46 
 47     //Zone folders
 48 
 49 
 50 
 51 
 52     for (let zoneName in missionOrgData.zones) {
 53         let zoneFolderData = allZoneFolderData[zoneName];
 54         let zoneFolderID = zoneFolderData.folder;
 55         let zoneFolder = DriveApp.getFolderById(zoneFolderID);
 56 
 57         let zoneOrgData = missionOrgData.zones[zoneName];
 58 
 59         if (DBCONFIG.LOG_FILE_SHARING) Logger.log(`Sharing folder for zone '${zoneName}'`);
 60 
 61 
 62         //      Update folder access: zEmails, officeEmails
 63 
 64 
 65         //List of emails with limited editor access at the zone level
 66         let zEmails = [];
 67         zEmails.push(zoneOrgData.zlArea.areaEmail);  //Add ZL area email
 68         if (zoneOrgData.hasStlArea) zEmails.push(zoneOrgData.stlArea.areaEmail);  //Add STL area email if it exists
 69 
 70         if (DBCONFIG.LOG_FILE_SHARING) Logger.log(`zEmails: ${zEmails}`);
 71 
 72         //Remove old editors, then add new ones
 73         let editorEmails = zoneFolder.getEditors().map(editor => { return editor.getEmail(); }); //Get a list of Editor objects, then convert to list of emails
 74         for (let editor of editorEmails) {
 75             if (!zEmails.includes(editor) && !officeEmails.includes(editor)) {
 76                 zoneFolder.removeEditor(editor);
 77             }
 78         }
 79 
 80         // zoneFolder.addEditors(zEmails);
 81         // zoneFolder.addEditors(officeEmails);
 82         silentAddEditors_(zoneFolderID, zEmails);
 83         silentAddEditors_(zoneFolderID, officeEmails);
 84 
 85         if (DBCONFIG.LOG_FILE_SHARING) Logger.log(`Removed and re-added zone folder editors: ${zoneFolder.getEditors().map(e => { return e.getName(); })}`);
 86 
 87 
 88         if (DBCONFIG.UPDATE_SHEET_PROTECTIONS_ON_FILESYS_LOAD) {
 89 
 90             //      Update Spreadsheet page protections: officeEmails only
 91 
 92             let zFiles = zoneFolder.getFilesByType("application/vnd.google-apps.spreadsheet"); //Get all the spreadsheet files in this folder
 93 
 94 
 95             while (zFiles.hasNext()) {  //for each spreadsheet file
 96                 let file = zFiles.next();
 97                 let ss = SpreadsheetApp.openById(file.getId());
 98                 let protections = ss.getProtections(SpreadsheetApp.ProtectionType.RANGE).concat(
 99                     ss.getProtections(SpreadsheetApp.ProtectionType.SHEET)
100                 );
101                 if (DBCONFIG.LOG_FILE_SHARING) Logger.log(`Updating ${protections.length} protections in sheet`);
102 
103                 for (let protection of protections) {
104                     for (let editor of protection.getEditors()) {
105                         if (!officeEmails.includes(editor)) {
106                             protection.removeEditor(editor);
107                         }
108                     }
109                     protection.addEditors(officeEmails);
110                 }
111             }
112 
113         }
114 
115 
116 
117 
118         //District folders (for districts within this zone)
119 
120 
121 
122 
123         for (let districtName in zoneOrgData.districts) {
124             let distFolderData = allDistFolderData[districtName];
125             let distFolderID = distFolderData.folder;
126             let distFolder = DriveApp.getFolderById(distFolderID);
127             let distOrgData = zoneOrgData.districts[districtName];
128 
129 
130             if (DBCONFIG.LOG_FILE_SHARING) Logger.log(`Sharing folder for district '${districtName}'`);
131 
132 
133             //      Update folder access: everyone with zone level access, plus the DL
134 
135 
136             //List of emails with limited editor access at the district level
137             let dEmails = [];
138             dEmails.push(distOrgData.dlArea.areaEmail);  //Add DL area email
139 
140             //Remove old editors (except for the ZLs and office emails), then add back the DL area email
141             let editorEmails = distFolder.getEditors().map(editor => { return editor.getEmail(); }); //Get a list of Editor objects, then convert to list of emails
142             for (let editor of editorEmails) {
143                 if (!dEmails.includes(editor)
144                     && !zEmails.includes(editor)
145                     && !officeEmails.includes(editor)) {
146                     distFolder.removeEditor(editor);
147                 }
148             }
149 
150             // distFolder.addEditors(dEmails);
151             silentAddEditors_(distFolderID, dEmails);
152 
153 
154 
155             if (DBCONFIG.UPDATE_SHEET_PROTECTIONS_ON_FILESYS_LOAD) {
156 
157                 //      Update Spreadsheet page protections: officeEmails only
158 
159                 let dFiles = distFolder.getFilesByType("application/vnd.google-apps.spreadsheet"); //Get all the spreadsheet files in this folder
160 
161                 while (dFiles.hasNext()) {  //for each spreadsheet file
162                     let ss = SpreadsheetApp.openById(dFiles.next().getId());
163                     let protections = ss.getProtections(SpreadsheetApp.ProtectionType.RANGE).concat(
164                         ss.getProtections(SpreadsheetApp.ProtectionType.SHEET)
165                     );
166                     if (DBCONFIG.LOG_FILE_SHARING) Logger.log(`Updating ${protections.length} protections in sheet`);
167 
168                     for (let protection of protections) {
169                         for (let editor of protection.getEditors()) {
170                             if (!officeEmails.includes(editor)) {
171                                 protection.removeEditor(editor);
172                             }
173                         }
174                         protection.addEditors(officeEmails);
175                     }
176                 }
177             }
178 
179 
180 
181 
182             //Area folders (for areas within this district)
183 
184 
185 
186 
187             for (let areaName in distOrgData.areas) {
188                 let areaFolderData = allAreaFolderData[areaName];
189                 let areaFolderID = areaFolderData.folder;
190                 let areaFolder = DriveApp.getFolderById(areaFolderID);
191                 let areaOrgData = distOrgData.areas[areaName];
192 
193 
194                 if (DBCONFIG.LOG_FILE_SHARING) Logger.log(`Sharing folder for area '${areaName}'`);
195 
196 
197                 //      Update folder access: everyone with zone or district level access, plus the area email
198 
199 
200                 //List of emails with limited editor access at the area level
201                 let aEmails = [];
202                 aEmails.push(areaOrgData.areaEmail);  //Add area email
203 
204                 //Remove old editors (except for the DL, ZLs, and office emails), then add back the DL area email
205                 let editorEmails = distFolder.getEditors().map(editor => { return editor.getEmail(); }); //Get a list of Editor objects, then convert to list of emails
206                 for (let editor of editorEmails) {
207                     if (!aEmails.includes(editor)
208                         && !dEmails.includes(editor)
209                         && !zEmails.includes(editor)
210                         && !officeEmails.includes(editor)) {
211                         areaFolder.removeEditor(editor);
212                     }
213                 }
214 
215                 areaFolder.addEditors(aEmails);
216                 silentAddEditors_(areaFolderID, aEmails);
217 
218 
219 
220                 if (DBCONFIG.UPDATE_SHEET_PROTECTIONS_ON_FILESYS_LOAD) {
221                     //      Update Spreadsheet page protections: officeEmails only
222 
223 
224                     let aFiles = areaFolder.getFilesByType("application/vnd.google-apps.spreadsheet"); //Get all the spreadsheet files in this folder
225 
226                     while (aFiles.hasNext()) {  //for each spreadsheet file
227                         let ss = SpreadsheetApp.openById(aFiles.next().getId());
228                         let protections = ss.getProtections(SpreadsheetApp.ProtectionType.RANGE).concat(
229                             ss.getProtections(SpreadsheetApp.ProtectionType.SHEET)
230                         );
231                         if (DBCONFIG.LOG_FILE_SHARING) Logger.log(`Updating ${protections.length} protections in sheet`);
232 
233                         for (let protection of protections) {
234                             for (let editor of protection.getEditors()) {
235                                 if (!officeEmails.includes(editor)) {
236                                     protection.removeEditor(editor);
237                                 }
238                             }
239                             protection.addEditors(officeEmails);
240                         }
241                     }
242                 }
243 
244 
245 
246 
247                 //Finished sharing folder for this area
248             }
249 
250             Logger.log(`Finished sharing folders for district ${districtName}`);
251         }
252 
253         Logger.log(`Finished sharing folders for zone ${zoneName}`);
254     }
255 
256 
257     Logger.log(`Completed read/write file sharing.`);
258 
259 
260 
261 
262 
263 
264 }

271 /**
272  * Adds the given user to the list of editors for the file or folder without sending a notification email.
273  * @param {string} fileId The file or folder ID.
274  * @param {string} emailAddress The email address of the user to add.
275  */
276 function silentAddEditor_(fileId, emailAddress) {
277 
278     Drive.Permissions.insert(
279         {
280             'role': 'writer',
281             'type': 'user',
282             'value': emailAddress
283         },
284         fileId,
285         {
286             'sendNotificationEmails': 'false'
287         });
288 
289 
290 }
291 
292 /**
293  * Adds the given array of users to the list of editors for the file or folder without sending notification emails.
294  * @param {string} fileId The ID of the file or folder to be edited.
295  * @param {string[]} emailAddresses An array of email addresses of the users to add.
296  */
297 function silentAddEditors_(fileId, emailAddresses) {
298     for (let email of emailAddresses)
299         silentAddEditor_(fileId, email);
300 }

269 function getParentFolderID_() {
270     // from https://stackoverflow.com/questions/17582161/google-apps-spreadsheet-parents-folder-id
271     let ss = SpreadsheetApp.getActiveSpreadsheet();
272     let ssid = ss.getId();
273     let fileInDrive = DriveApp.getFolderById(ssid);
274     let folderinDrive = fileInDrive.getParents().next().getId();
275     // Logger.log(folderinDrive)
276     return folderinDrive;
277 }

294 function getSheetOrSetUp_(sheetName, headerData) {
295 
296     let ss; //Get currently Active sheet
297     ss = SpreadsheetApp.getActiveSpreadsheet();
298 
299     // Checks to see if the sheet exists or not.
300     let sheet = ss.getSheetByName(sheetName);
301     if (!sheet) {
302         sheet = ss.insertSheet(sheetName);
303         sheet.appendRow(headerData);// Creating Header
304     }
305     return sheet;

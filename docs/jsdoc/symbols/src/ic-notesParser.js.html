  2 function parseNotes(inputString) {
  3     // PARSER FOR NOTES SECTION DATA
  4     if (typeof (inputString) != 'string') {
  5         TypeError("NO NOTES FOR NOTES PARSER TO PARSE.  PLZ HALP, HAVING EXISTENTIAL CRISIS");
  6     }
  7 
  8     var notesSplit = inputString.split("\n");
  9     var hasVehicle = inputString.includes("Vehicle");
 10 
 11     var area = notesSplit[0].replace("Area: ", "").trim();
 12 
 13     if (DBCONFIG.LOG_IMPORT_CONTACTS) Logger.log(`Parsing notes for Contact: ${area}`);
 14 
 15     var district = notesSplit[1].replace("District: ", "").trim();
 16     var zone = notesSplit[2].replace("Zone: ", "").trim();
 17     let isSeniorCouple = notesSplit[1].includes("Senior");
 18     var ecclesiasticalUnitString = notesSplit[3];
 19     var hasMultipleUnits = ecclesiasticalUnitString.includes("Units:"); // tests to see if there are multiple ecclesiastical units covered by a companionship.
 20     var unitString = "";
 21     if (hasMultipleUnits == true) {
 22         unitString = ecclesiasticalUnitString.replace("Ecclesiastical Units:", "").trim();
 23     }
 24     if (hasMultipleUnits == false) {
 25         unitString = ecclesiasticalUnitString.replace("Ecclesiastical Unit:", "").trim();
 26         unitString = unitString.replace("EcclesiasticalUnit:", "").trim();
 27     }
 28 
 29     // This part is in a conditional because it only matters (& WORKS!) if there's a vehicle.
 30     // Otherwise, the vehicle will get the extra tags and junk, which we don't particularly want to happen.
 31 
 32     var vehicleDesc;
 33     var vehicleMiles;
 34     var vinLast8;
 35     var vehicleCCID;
 36     var finalTags;
 37 
 38     if (hasVehicle == true) {
 39         vehicleDesc = notesSplit[5].replace("Vehicle:", "").trim();
 40         vehicleMiles = notesSplit[6].replace("Vehicle Allowance/Mo: ", "").replace("Mi", "").trim();
 41         vinLast8 = notesSplit[7].replace("Vehicle VIN Last 8:", "").trim();
 42         vehicleCCID = notesSplit[8].replace("Vehicle", "").replace("CCID:", "").trim();
 43 
 44         // This has *some* data in it but I'm not using it right now
 45         finalTags = notesSplit[10];
 46     } else {
 47         finalTags = notesSplit[5];
 48     }
 49 
 50     let isSisterArea = finalTags.includes("Sister");
 51 
 52     return {
 53         area: area,
 54         district: district,
 55         zone: zone,
 56         hasMultipleUnits: hasMultipleUnits,
 57         unitString: unitString,
 58         hasVehicle: hasVehicle,
 59         vehicleDesc: vehicleDesc,
 60         vehicleMiles: vehicleMiles,
 61         vinLast8: vinLast8,
 62         vehicleCCID: vehicleCCID,
 63         isSeniorCouple: isSeniorCouple,
 64         isSisterArea: isSisterArea,
 65         finalTags: finalTags
 66     };
 67 } // Written by Elder Robertson, TMM

 18 
 19 missionData = [
 20 
 21   //form response 1
 22   {
 23     "metadata" : { //metadata about the response, as opposed to area data (includes areaID). These WILL NOT appear in the Data sheet unless specified to do so.

 25       "hasContactData" : TRUE,
 26       "dateContactImported" : "11/7/2021",
 27       ...
 28     },
 29 
 30     "areaData" : { //data about the area, as opposed to metadata (does not include areaID). These will all appear in the Data sheet.
 31       "areaName" : "B1A", "np" : 4, ...
 32     }
 33 
 34   },
 35 
 36   //form response 2
 37   {
 38 
 39   }...
 40 ]
 41 
 42 missionData[3].areaData.areaName => "B1A"
 43 missionData[3].metadata.hasContactData => true
 44 missionData[3].metadata.areaID => "123456@missionary.org"

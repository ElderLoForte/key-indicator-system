126 /**
127  * Returns true if Contact Data was not imported recently and needs to be refreshed.
128  */
129 function isContactDataOld(allSheetData) {
130     let msOffset = 86400000; //Number of milliseconds in 24 hours
131     let nowTime = new Date();
132     let genTime = allSheetData.contact.getSheet().getRange("A2").getValue();
133     let out;
134 
135     try {
136         out = genTime.getTime() < nowTime.getTime() - msOffset;
137     } catch (e) {
138         out = true;
139     }
140 
141     return out;
142 }

166 /**
167  * Parses email data for importContacts()
168  */
169 function emailParser(emailList) {
170     let emailAddresses = [];
171     let emailDisplayNames = [];
172     let emailLabelName = [];
173     // this operates under the assumption that all the emails are in the same order :0
174     for (let i = 0; i < emailList.length; i++) {
175         emailAddresses[i] = emailList[i].getAddress();
176         emailDisplayNames[i] = emailList[i].getDisplayName();
177         emailLabelName[i] = emailList[i].getLabel();
178     }
179     return {
180         emailAddresses: emailAddresses,
181         emailDisplayNames: emailDisplayNames,
182         emailLabelNames: emailLabelName
183     };
184 }

203 /**
204  * Parses language data for importContacts()
205  */
206 function languageParser(multipleUnits, unitString) {
207 
208     // noteData.UnitString.substring(noteData.UnitString.search(/\(\w*/))
209     let defaultLanguage = "English";
210     let spanishTestString = "Spanish";
211     let spanishOutputString = "Spanish";
212     let returnData = [];
213 
214     if (!multipleUnits) {
215         if (unitString.includes(spanishTestString) == true) // this is going to get changed in the future to get rid of the silly 'spanish,spanish' tags.
216             returnData.push(spanishOutputString);
217         else
218             returnData.push(defaultLanguage);
219 
220     } else {
221         let unitStringSplit = unitString.split(",");
222 
223         for (let testString of unitStringSplit) {
224             if (testString.includes(spanishTestString))
225                 returnData.push(spanishOutputString);
226             else
227                 returnData.push(defaultLanguage);
228         }
229     }
230 
231     return { 'languageString': returnData.toString() };
232 }

 79 function dataLoader_(allSheetData, scope) {
 80     let sheetDataClass;
 81 
 82     switch (scope) {
 83         case reportLevel.zone:
 84             sheetDataClass = allSheetData.zoneFilesys;
 85             break;
 86         case reportLevel.dist:
 87             sheetDataClass = allSheetData.distFilesys;
 88             break;
 89         case reportLevel.area:
 90             sheetDataClass = allSheetData.areaFilesys;
 91     }
 92     let sheetObj = sheetDataClass.getSheet();
 93     // split will have to get changed once Elder Gerlek updates his chunks; headerSplit_ won't need to exist for much longer.
 94     let split = headerSplit_(sheetObj.getDataRange().getValues());
 95     let fsObj = loadFSIntoClass_(split.data, sheetDataClass);
 96 
 97     return {
 98         sheetData: sheetDataClass,
 99         sheet: sheetObj,
100         splitData: split,
101         fsObj: fsObj,
102     };
103 }

335 /*
336  * @param {any} folderID
337  */
338 function isFolderAccessible_(folderID) {
339     // This just try catches to see if there's a folder, because for some reason this is the most effective way to do it...
340     let output = true;
341     let folder;
342     let gone = false;
343     try {
344         folder = DriveApp.getFolderById(folderID);
345         // test = DriveApp.getFolderById(folderID).getName();
346 
347         folder.getDescription();
348     } catch (e) {
349         output = false;
350         gone = true;
351         Logger.log("Folder deleted with ID " + folderID);
352     }
353     if (gone == false) {
354         if (folder.isTrashed() == true) {
355             Logger.log("folder exists but in the bin");
356             output = false;
357         }
358     }
359 
360     return output;
361 }

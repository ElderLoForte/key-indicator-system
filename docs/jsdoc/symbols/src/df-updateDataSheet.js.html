 17 /**
 18   * Updates the Data sheet.
 19   */
 20 function updateDataSheet() {
 21     Logger.log("BEGINNING UPDATE");
 22 
 23     let allSheetData = constructSheetData();
 24     if (CONFIG.FORCE_AREA_ID_RELOAD_ON_UPDATE_DATA_SHEET) loadAreaIDs(allSheetData); //Force a full recalculation
 25 
 26     //checkForErrors()?  Ex. no contact data
 27 
 28     let missionData = pullFormData(allSheetData);
 29 
 30     if (missionData.length == 0) {
 31         Logger.log("UPDATE COMPLETED - NO NEW FORM RESPONSES FOUND");
 32         return;
 33     }
 34 
 35     refreshContacts(allSheetData);
 36 
 37     let contacts = getContactData(allSheetData);
 38 
 39     let leaders = getLeadershipAreaData(contacts);
 40 
 41     missionData = mergeIntoMissionData(missionData, contacts, "contact data");
 42     missionData = mergeIntoMissionData(missionData, leaders, "leadership data");
 43 
 44 
 45     allSheetData.data.insertData(missionData);
 46 
 47     markDuplicates(allSheetData);
 48 
 49     pushErrorMessages();  //Unimplemented
 50 
 51     Logger.log("UPDATE COMPLETED");
 52 }

 65 /**
 66   * Pulls data from the Form Response sheet and adds areaIDs. Hard-codes column order for the initial columns, and pulls later columns automatically, using the values in the header row as keys.
 67   */
 68 function pullFormData(allSheetData) {
 69     Logger.log("Pulling Form Data...");
 70 
 71     let fSheetData = allSheetData.form;
 72     let responses = fSheetData.getData();
 73     let missionData = [];
 74 
 75     Logger.log("[TODO] Limit pullFormData from pulling the whole sheet - sheetData.getRecentData(maxRows) or something similar? Specify max and min rows?");
 76 
 77 
 78     for (let response of responses) {
 79         if (response.responsePulled == true || response.areaName == "")
 80             continue;
 81 
 82         if (DBCONFIG.LOG_RESPONSE_PULLED) Logger.log("Pulling response for area: '" + response.areaName + "'");
 83 
 84         response.areaID = getAreaID(allSheetData, response.areaName);
 85 
 86         response.log =
 87         {
 88             'areaID': response.areaID,
 89             'areaName': response.areaName,
 90             'processDate': (new Date()).toDateString(),
 91             'formDataPulled': true,
 92             'formDataPulledTime': (new Date()).toTimeString(),
 93             'formTimestamp': response.formTimestamp,
 94         };
 95 
 96         missionData.push(response);
 97     }
 98 
 99 
100     //Mark responses as having been pulled
101     console.info("TODO: Improve marking responses as pulled");
102     if (DBCONFIG.SKIP_MARKING_PULLED) {
103         Logger.log("[DEBUG] Skipping marking Form Responses as having been pulled into the data sheet: SKIP_MARKING_PULLED is set to true");
104     }
105     else {
106         let formSheet = fSheetData.getSheet();
107         let markerRange = formSheet.getRange("B2:B" + formSheet.getLastRow());
108         formSheet.getRange("B2").setValue(true);
109         formSheet.getRange("B2").autoFill(markerRange, SpreadsheetApp.AutoFillSeries.DEFAULT_SERIES);
110     }
111 
112     Logger.log("Finished pulling Form Data.");
113     return missionData;
114 }

124 /**
125   * Pulls data from the Contact Data sheet and adds areaIDs.
126   */
127 function getContactData(allSheetData) {
128 
129     Logger.log("Getting data from Contact Data sheet...");
130 
131     let cSheetData = allSheetData.contact;
132     let contactData = cSheetData.getData();
133     let contacts = {}; //contactData, keyed by areaID
134     for (let contact of contactData) {
135         contact.areaID = getAreaID(allSheetData, contact.areaName);
136         if ((typeof contact.log) == 'undefined')
137             contact.log = {};
138         contact.log.addedContactData = true;
139         contact.log.addedContactDataTime = (new Date()).toTimeString();
140 
141         if ((typeof contacts[contact.areaID]) != 'undefined')
142             warnDataCollision(contact.areaName, contact.areaID, contacts[contact.areaID].areaName);
143 
144         contacts[contact.areaID] = contact;
145     }
146 
147     Logger.log("Finished pulling contact data.");
148 
149     return contacts;
150 
151 
152     function warnDataCollision(area, id, otherArea) {
153         console.warn("Potential data collision while pulling data from contacts: tried to add area '" + area + "' with id '" + id + "', but that id already has data for area '" + otherArea + "'");
154     }
155 }

159 /**
160   * Merges data from an additional data source (contact data, config data, etc) into missionData, assuming the formatting Reference.gs describes.
161   * Used to pull contact and leader data into missionData.
162   * Takes a reference to missionData, a reference to the datasource, and an ID string for that datasource.
163   */
164 function mergeIntoMissionData(missionData, sourceData, sourceID) {
165     Logger.log("Beginning to merge source '" + sourceID + "' into missionData");
166 
167     let newMissionData = [];
168     let mdKeys = Object.keys(missionData[0]);
169     let sdKeys = Object.keys(sourceData[missionData[0].areaID]);
170     let keys = new Set(mdKeys.concat(sdKeys)); //Set of all keys from both objects (a Set removes duplicates automatically)
171 
172 
173     for (let missionAreaData of missionData) {
174         let areaID = missionAreaData.areaID;
175         let areaName = missionAreaData.areaName;
176         let sourceAreaData = sourceData[missionAreaData.areaID];
177 
178         if (DBCONFIG.LOG_MERGE_DATA) Logger.log("Merging area '" + areaName + "' (id '" + areaID + "') from source " + sourceID);
179 
180         if (typeof sourceAreaData == 'undefined') //Error if can't find corresponding areaID
181             throw "Found a form response for area '" + areaName + "' (id '" + areaID + "'), but couldn't find that area in source '" + sourceID + "'";
182 
183 
184         let newAreaData = {};
185         let mergeLog =
186         {
187             'missingKeys': [],
188             'collisions': {},
189         };
190 
191         for (let key of keys) {
192 
193             let mHasKey = typeof missionAreaData[key] != 'undefined';
194             let sHasKey = typeof sourceAreaData[key] != 'undefined';
195 
196             //Log warnings if neither object has this key (should be unreachable), or if both do and they disagree
197             if (!mHasKey && !sHasKey) {
198                 logNeither(key, areaID, areaName);
199                 mergeLog.missingKeys.push(key);
200             }
201             else if (mHasKey && sHasKey && missionAreaData[key] != sourceAreaData[key]) {
202                 logDataCollision(key, areaID, areaName, sourceID, sourceAreaData[key], missionAreaData[key]);
203                 mergeLog.collisions[key] =
204                 {
205                     'missionVal': missionAreaData[key],
206                     'sourceVal': sourceAreaData[key]
207                 };
208 
209             }
210 
211             //Set new value
212             //Neither: empty string; missionData only: missionData value; sourceData only (or both): source value
213             newAreaData[key] = (!mHasKey && !sHasKey) ? "" : (mHasKey && !sHasKey) ? missionAreaData[key] : sourceAreaData[key];
214 
215         }
216 
217         if (typeof newAreaData.log == 'undefined') newAreaData.log = {};
218         if (typeof newAreaData.log.merges == 'undefined') newAreaData.log.merges = {};
219         newAreaData.log.merges[sourceID] = mergeLog;
220 
221         newMissionData.push(newAreaData);
222     }
223 
224     Logger.log("Finished merging source '" + sourceID + "'");
225 
226     return newMissionData;
227 
228 
229 
230     function logNeither(key, areaID, areaName, sourceID) {
231         console.warn("Warning: couldn't find key '"+key+"' for area '"+areaName+"' (id '"+areaID+"') in either mission data or source '"+source+"'");
232     }
233 
234     function logDataCollision(key, areaID, areaName, sourceID, sourceAreaDataOfKey, missionAreaDataOfKey) {
235         Logger.log("Warning: possible data collision on key '" + key + "' for area '" + areaName + "' (id '" + areaName + "'). Source '" + sourceID + "' has value '" + sourceAreaDataOfKey + "' while missionData has value '" + missionAreaDataOfKey + "'");
236     }
237 
238 }
